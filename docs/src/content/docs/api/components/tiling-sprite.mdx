Renders a PIXI.TilingSprite.

<ComponentDescription name="TilingSprite" source="https://pixijs.download/dev/docs/scene.TilingSprite.html" />


## Usage

<SvelteTabs>
  <div slot="svelte-5">
    ```svelte live props={{ assets: ['/assets/tiling-sprite.jpeg'] }}
    <script>
      import { TilingSprite, onTick } from 'svelte-pixi'
      import * as PIXI from 'pixi.js'

      let scale = $state({ x: 1, y: 1 })
      let tilePosition = $state({ x: 0, y: 0 })

      onTick((ticker) => {
        const time = ticker.lastTime / 2000

        scale = { x: 2 + Math.sin(time), y: 2 + Math.cos(time) }
        tilePosition = { x: tilePosition.x + 1, y: tilePosition.y + 1 }
      })
    </script>

    <TilingSprite
      x={360}
      y={200}
      texture={PIXI.Texture.from('/assets/tiling-sprite.jpeg')}
      anchor={0.5}
      width={400}
      height={400}
      {tilePosition}
      {scale}
    />
    ```
  </div>
  <div slot="svelte-4">

  </div>
