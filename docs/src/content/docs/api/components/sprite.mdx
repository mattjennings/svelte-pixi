Renders a PIXI.Sprite. Textures **must** be loaded before use.

<ComponentDescription name="Sprite" source="https://pixijs.download/dev/docs/scene.Sprite.html" />

## Usage

<SvelteTabs>
  <div slot="svelte-5">
    ```svelte live props={{ assets: ['/assets/adventurer.png'] }}
    <script>
      import * as PIXI from 'pixi.js'
      import { Sprite } from 'svelte-pixi/svelte-5'
    </script>

    <Sprite
      x={360}
      y={200}
      texture={PIXI.Texture.from('/assets/adventurer.png')}
      anchor={0.5}
      scale={3}
    />
    ```
  </div>
    <div slot="svelte-4">
    ```svelte live props={{ assets: ['/assets/adventurer.png'] }}

      import { Sprite } from 'svelte-pixi/svelte-4'
    </script>

    <Sprite
      x={360}
      y={200}
      texture={PIXI.Texture.from('/assets/adventurer.png')}
      anchor={0.5}
      scale={3}
    />
    ```
  </div>
</SvelteTabs>
