Renders a PIXI.BitmapText. Bitmap fonts **must** be loaded before use.

<ComponentDescription
  name="BitmapText"
  source="https://pixijs.download/dev/docs/scene.BitmapText.html"
/>

## Usage

<SvelteTabs>
  <div slot="svelte-5">
    ```svelte live props={{ bg: 'white' }}
    <script>
      import { AssetsLoader, BitmapText } from 'svelte-pixi'
    </script>

    <AssetsLoader assets={['/assets/desyrel.xml']}>
      <BitmapText
        text={'the fox jumps over\nthe lazy dog'}
        x={380}
        y={200}
        anchor={0.5}
        style={{ fontFamily: 'Desyrel', fontSize: 48, align: 'center' }}
      />
    </AssetsLoader>
    ```

  </div>
  <div slot="svelte-4">
    ```svelte live props={{ bg: 'white' }}
    <script>
      import { AssetsLoader, BitmapText } from 'svelte-pixi/svelte-4'
    </script>

    <AssetsLoader assets={['/assets/desyrel.xml']}>
      <BitmapText
        text={'the fox jumps over\nthe lazy dog'}
        x={380}
        y={200}
        anchor={0.5}
        style={{ fontFamily: 'Desyrel', fontSize: 48, align: 'center' }}
      />
    </AssetsLoader>
    ```

  </div>
</SvelteTabs>
