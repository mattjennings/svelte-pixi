### Updating style

<SvelteTabs>
  <div slot="svelte-5">
    ```svelte live props={{ assets: ['/assets/adventurer.png'] }}
    <script>
      import * as PIXI from 'pixi.js'
      import { Text, tick } from 'svelte-pixi'

      let fill = tick(t => {
        const time = t.lastTime
        const r = Math.floor(Math.sin(time / 1000) * 128 + 128)
        const g = Math.floor(Math.sin(time / 500) * 128 + 128)
        const b = Math.floor(Math.sin(time / 300) * 128 + 128)
        
        return (r << 16) + (g << 8) + b
      })
    </script>

    <Text
      x={300}
      y={200}
      text="Hello world"
      style={{ fill: $fill }}
    />
    ```
  </div>
    <div slot="svelte-4">
    ```svelte live props={{ assets: ['/assets/adventurer.png'] }}
    <script>
      import * as PIXI from 'pixi.js'
      import { Text, tick } from 'svelte-pixi/svelte-4'
      
      let fill = tick(t => {
        const time = t.lastTime
        const r = Math.floor(Math.sin(time / 1000) * 128 + 128)
        const g = Math.floor(Math.sin(time / 500) * 128 + 128)
        const b = Math.floor(Math.sin(time / 300) * 128 + 128)
        
        return (r << 16) + (g << 8) + b
      })
    </script>

    <Text
      x={300}
      y={200}
      text="Hello world"
      style={{ fill: $fill }}
    />
    ```
  </div>
</SvelteTabs>
